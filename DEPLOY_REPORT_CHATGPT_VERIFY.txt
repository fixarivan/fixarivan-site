--------------------------------------------------------------------------------
1. НАЗНАЧЕНИЕ
--------------------------------------------------------------------------------

Bash-скрипт production-уровня для выкладки PHP-проекта FixariVan на shared-
хостинг по SSH: локальная валидация, бэкап на сервере, очистка каталога сайта,
передача файлов (tar + scp + распаковка), выставление прав, откат из архива
с промежуточной распаковкой во временный каталог (безопасность).

Запуск: Git Bash / Linux / WSL (путь к SSH-ключу указан в формате Git Bash).

--------------------------------------------------------------------------------
2. ЗАВИСИМОСТИ
--------------------------------------------------------------------------------

  - bash 4+
  - ssh, scp, tar, find, xargs
  - php (для validate: php -l)
  - На сервере для отката желателен rsync; иначе используется cp -a

--------------------------------------------------------------------------------
3. КОНФИГУРАЦИЯ (КОНСТАНТЫ В СКРИПТЕ)
--------------------------------------------------------------------------------

  SCRIPT_DIR     — корень проекта (авто: каталог deploy.sh)
  REMOTE_USER    — fixawcab
  REMOTE_HOST    — 162.0.217.157
  REMOTE_PORT    — 21098
  SSH_KEY        — /c/Users/user/Downloads/id_rsa
  REMOTE_PATH    — /home/fixawcab/public_html/fixarivan.space
  BACKUP_PATH    — /home/fixawcab/backups
  TMP_RESTORE    — /home/fixawcab/tmp_restore
  SITE_PARENT    — /home/fixawcab/public_html
  SITE_NAME      — fixarivan.space

--------------------------------------------------------------------------------
4. АРГУМЕНТЫ
--------------------------------------------------------------------------------

  (без аргументов) — полный деплой после проверок
  --dry             — dry-run: validate и «симуляция» шагов без разрушительных
                      действий на сервере (бэкап/clean/upload/finalize не
                      выполняются по сути; check_connection в dry не бьёт SSH)
  --rollback        — только откат: validate_remote_dir_constant + rollback()
                      (validate PHP для отката не вызывается — по выбранной
                      схеме main)

  Обрабатывается только ПЕРВЫЙ аргумент ($1).

--------------------------------------------------------------------------------
5. ФУНКЦИИ (КРАТКО)
--------------------------------------------------------------------------------

  parse_args()
    Устанавливает DRY_RUN или ROLLBACK.

  validate_remote_dir_constant()
    Жёсткая проверка, что REMOTE_PATH равен ожидаемому абсолютному пути.

  validate()
    - Сообщение: "Validating project..."
    - index.php в SCRIPT_DIR, php в PATH
    - Каталог api/ обязателен (иначе ошибка)
    - Отсутствие storage/ — предупреждение в stderr, не fail
    - Рекурсивно все *.php: php -l; при любой ошибке — вывод файла и вывода,
      failed=1, в конце return 1
    - Успех: "Validation OK"
    Выполняется и в режиме --dry.

  check_connection()
    ssh … "echo connected"; вывод сравнивается с "connected" (после tr -d \r).
    В --dry: только текст симуляции, без реального SSH.

  backup()
    На сервере: mkdir BACKUP_PATH, tar -czf BACKUP_PATH/fixarivan_YYYY-MM-DD_
    HH-MM-SS.tar.gz -C SITE_PARENT SITE_NAME. В dry — описание действия.

  clean()
    Внутри REMOTE_PATH: find mindepth/maxdepth 1, удалить всё кроме storage/,
    .env, backups/. Сам REMOTE_PATH не удаляется.

  upload()
    Локально: tar с исключениями → scp в /tmp/ → на сервере tar -xzf в
    REMOTE_PATH. scp не поддерживает exclude — поэтому сначала tar.

  finalize()
    chmod: каталоги 755, файлы 644; storage 755. Сообщение "Deploy completed
    successfully" только при реальном выполнении (не в dry finalize).

  rollback()
    - SSH-проверка как выше
    - Последний архив: ls -t fixarivan_*.tar.gz в BACKUP_PATH (nullglob)
    - Распаковка в TMP_RESTORE; при ошибке или отсутствии fixarivan.space —
      выход, продакшен не трогается (кроме очистки tmp при части ошибок)
    - Успех: очистка содержимого REMOTE_PATH, rsync или cp из TMP_RESTORE/
      fixarivan.space/ в REMOTE_PATH/, очистка TMP_RESTORE
    - "Rollback complete"

--------------------------------------------------------------------------------
6. ПОТОК main
--------------------------------------------------------------------------------

  parse_args
  если --rollback:
      validate_remote_dir_constant → rollback → return
  иначе:
      validate_remote_dir_constant → validate
      если --dry: сообщение "DRY RUN MODE"
      check_connection → backup → clean → upload → finalize
      если --dry: "Dry run finished (no remote changes)."

--------------------------------------------------------------------------------
7. ЦВЕТА И ОШИБКИ
--------------------------------------------------------------------------------

  ok/step/err — зелёный / жёлтый / красный; NO_COLOR отключает ANSI.
  При неуспехе main: "Operation failed."

--------------------------------------------------------------------------------
8. ИЗВЕСТНЫЕ ОГРАНИЧЕНИЯ (ДЛЯ РЕВЬЮ)
--------------------------------------------------------------------------------

  - Строгое сравнение вывода SSH с "connected" может ломаться при MOTD/лишних
    строках на сервере.
  - php -l по всем *.php может быть долгим (vendor).
  - Второй и последующие аргументы командной строки игнорируются.

--------------------------------------------------------------------------------
9. ПОЛНЫЙ ИСХОДНЫЙ КОД deploy.sh
--------------------------------------------------------------------------------
(Ниже — точная копия файла deploy.sh на момент формирования отчёта)

BEGIN deploy.sh
================================================================================

#!/usr/bin/env bash
#
# FixariVan — production deploy (bash, SSH, tar, scp).
# Usage: ./deploy.sh | ./deploy.sh --dry | ./deploy.sh --rollback
#

set -e
set -o pipefail

# =============================================================================
# CONFIG
# =============================================================================

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
REMOTE_USER="fixawcab"
REMOTE_HOST="162.0.217.157"
REMOTE_PORT="21098"
SSH_KEY="/c/Users/user/Downloads/id_rsa"

REMOTE_PATH="/home/fixawcab/public_html/fixarivan.space"
BACKUP_PATH="/home/fixawcab/backups"
TMP_RESTORE="/home/fixawcab/tmp_restore"
SITE_PARENT="/home/fixawcab/public_html"
SITE_NAME="fixarivan.space"

DRY_RUN=0
ROLLBACK=0
ARCHIVE=""

# =============================================================================
# Colors (NO_COLOR respected)
# =============================================================================

if [[ -t 1 ]] && [[ -z "${NO_COLOR:-}" ]]; then
  _G='\033[0;32m'
  _R='\033[0;31m'
  _Y='\033[1;33m'
  _N='\033[0m'
else
  _G='' _R='' _Y='' _N=''
fi

ok() { printf '%b%s%b\n' "$_G" "$*" "$_N"; }
err() { printf '%b%s%b\n' "$_R" "$*" "$_N" >&2; }
step() { printf '%b%s%b\n' "$_Y" "$*" "$_N"; }

# =============================================================================
# SSH helper
# =============================================================================

# shellcheck disable=SC2206
SSH_BASE=( ssh -i "$SSH_KEY" -p "$REMOTE_PORT" -o IdentitiesOnly=yes -o BatchMode=yes -o ConnectTimeout=15 )

ssh_run() {
  "${SSH_BASE[@]}" "${REMOTE_USER}@${REMOTE_HOST}" "$@"
}

# =============================================================================
# Functions
# =============================================================================

parse_args() {
  case "${1:-}" in
    --dry)
      DRY_RUN=1
      ;;
    --rollback)
      ROLLBACK=1
      ;;
    "")
      ;;
    *)
      err "Unknown option: ${1:-} (use --dry or --rollback)"
      exit 1
      ;;
  esac
}

validate_remote_dir_constant() {
  [[ "$REMOTE_PATH" == "/home/fixawcab/public_html/fixarivan.space" ]] || {
    err "Refusing: REMOTE_PATH safety check failed."
    return 1
  }
}

validate() {
  step "Validating project..."

  if [[ ! -f "${SCRIPT_DIR}/index.php" ]]; then
    err "Validation failed: index.php not found in ${SCRIPT_DIR}"
    return 1
  fi

  if ! command -v php >/dev/null 2>&1; then
    err "Validation failed: php not found in PATH"
    return 1
  fi

  if [[ ! -d "${SCRIPT_DIR}/api" ]]; then
    err "Validation failed: api/ directory is missing"
    return 1
  fi

  if [[ ! -d "${SCRIPT_DIR}/storage" ]]; then
    err "Warning: storage/ directory is missing locally (optional on dev machine)" >&2
  fi

  local failed=0
  local f out
  while IFS= read -r -d '' f; do
    if ! out="$(php -l "$f" 2>&1)"; then
      err "Validation failed: ${f}"
      printf '%s\n' "$out" >&2
      failed=1
    fi
  done < <(find "$SCRIPT_DIR" -type f -name '*.php' -print0)

  if [[ "$failed" -ne 0 ]]; then
    err "Validation failed: fix PHP syntax errors above."
    return 1
  fi

  ok "Validation OK"
}

check_connection() {
  step "Checking SSH connection..."
  if [[ "$DRY_RUN" -eq 1 ]]; then
    step "[dry-run] would run: ssh … \"echo connected\""
    return 0
  fi

  local out
  out="$(ssh_run "echo connected" 2>/dev/null | tr -d '\r' || true)"
  if [[ "$out" != "connected" ]]; then
    err "SSH check failed (expected output \"connected\", got: ${out:-<empty>})"
    return 1
  fi
  ok "SSH connection OK"
}

backup() {
  step "Creating backup..."
  if [[ "$DRY_RUN" -eq 1 ]]; then
    step "[dry-run] would create: ${BACKUP_PATH}/fixarivan_YYYY-MM-DD_HH-MM-SS.tar.gz (full ${SITE_NAME})"
    return 0
  fi

  validate_remote_dir_constant
  ssh_run bash -s <<'REMOTE_BACKUP'
set -euo pipefail
BACKUP_PATH="/home/fixawcab/backups"
SITE_PARENT="/home/fixawcab/public_html"
SITE_NAME="fixarivan.space"
[[ "$BACKUP_PATH" == "/home/fixawcab/backups" ]] || exit 1
[[ "$SITE_PARENT" == "/home/fixawcab/public_html" ]] || exit 1
mkdir -p "$BACKUP_PATH"
TS="$(date +%Y-%m-%d_%H-%M-%S)"
OUT="${BACKUP_PATH}/fixarivan_${TS}.tar.gz"
if [[ ! -d "${SITE_PARENT}/${SITE_NAME}" ]]; then
  echo "Site directory missing: ${SITE_PARENT}/${SITE_NAME}" >&2
  exit 1
fi
tar -czf "$OUT" -C "$SITE_PARENT" "$SITE_NAME"
printf '%s\n' "$OUT"
REMOTE_BACKUP
  ok "Backup created (see path above on server)."
}

clean() {
  step "Cleaning server..."
  if [[ "$DRY_RUN" -eq 1 ]]; then
    step "[dry-run] would: find inside REMOTE_PATH, delete all except storage/, .env, backups/"
    return 0
  fi

  validate_remote_dir_constant
  ssh_run bash -s <<EOF
set -euo pipefail
TARGET=$(printf '%q' "$REMOTE_PATH")
[[ "\$TARGET" == "/home/fixawcab/public_html/fixarivan.space" ]] || exit 1
[[ -d "\$TARGET" ]] || exit 1
find "\$TARGET" -mindepth 1 -maxdepth 1 ! -name 'storage' ! -name '.env' ! -name 'backups' -print0 | xargs -0 rm -rf
EOF
}

upload() {
  step "Uploading files..."
  if [[ "$DRY_RUN" -eq 1 ]]; then
    step "[dry-run] would: tar (exclude .git, node_modules, *.log, .DS_Store) → scp → extract into ${REMOTE_PATH}"
    return 0
  fi

  validate_remote_dir_constant
  ARCHIVE="${SCRIPT_DIR}/.fixarivan_deploy_$$.tar.gz"
  trap 'rm -f "${ARCHIVE}"' EXIT

  tar -czf "$ARCHIVE" \
    --exclude='.git' \
    --exclude='node_modules' \
    --exclude='*.log' \
    --exclude='.DS_Store' \
    --exclude='.fixarivan_deploy_*.tar.gz' \
    -C "$SCRIPT_DIR" .

  REMOTE_ARCHIVE="fixarivan_upload_$(date +%s)_$$.tar.gz"
  scp -i "$SSH_KEY" -P "$REMOTE_PORT" -o IdentitiesOnly=yes -o BatchMode=yes "$ARCHIVE" "${REMOTE_USER}@${REMOTE_HOST}:/tmp/${REMOTE_ARCHIVE}"

  ssh_run bash -s <<EOF
set -euo pipefail
TARGET=$(printf '%q' "$REMOTE_PATH")
ARCH=$(printf '%q' "/tmp/${REMOTE_ARCHIVE}")
[[ "\$TARGET" == "/home/fixawcab/public_html/fixarivan.space" ]] || exit 1
[[ -f "\$ARCH" ]] || exit 1
[[ -d "\$TARGET" ]] || exit 1
tar -xzf "\$ARCH" -C "\$TARGET"
rm -f "\$ARCH"
EOF

  rm -f "$ARCHIVE"
  trap - EXIT
}

finalize() {
  if [[ "$DRY_RUN" -eq 1 ]]; then
    step "[dry-run] would: chmod dirs 755, files 644; chmod 755 storage"
    return 0
  fi

  validate_remote_dir_constant
  ssh_run bash -s <<EOF
set -euo pipefail
TARGET=$(printf '%q' "$REMOTE_PATH")
[[ "\$TARGET" == "/home/fixawcab/public_html/fixarivan.space" ]] || exit 1
[[ -d "\$TARGET" ]] || exit 1
if [[ -d "\${TARGET}/storage" ]]; then
  chmod 755 "\${TARGET}/storage"
fi
find "\$TARGET" -type d -exec chmod 755 {} +
find "\$TARGET" -type f -exec chmod 644 {} +
EOF
  ok "Deploy completed successfully"
}

rollback() {
  step "Rolling back..."

  if [[ "$DRY_RUN" -eq 1 ]]; then
    step "[dry-run] would: check SSH, pick latest backup, extract to ${TMP_RESTORE}, then replace ${REMOTE_PATH} only if extract OK"
    ok "Rollback complete (dry run — no changes)"
    return 0
  fi

  step "Checking SSH connection..."
  local out
  out="$(ssh_run "echo connected" 2>/dev/null | tr -d '\r' || true)"
  if [[ "$out" != "connected" ]]; then
    err "SSH check failed — aborting rollback."
    return 1
  fi
  ok "SSH connection OK"

  ssh_run bash -s <<'REMOTE_RB'
set -euo pipefail
BACKUP_PATH="/home/fixawcab/backups"
TMP_RESTORE="/home/fixawcab/tmp_restore"
REMOTE_PATH="/home/fixawcab/public_html/fixarivan.space"
[[ "$BACKUP_PATH" == "/home/fixawcab/backups" ]] || exit 1
[[ "$TMP_RESTORE" == "/home/fixawcab/tmp_restore" ]] || exit 1
[[ "$REMOTE_PATH" == "/home/fixawcab/public_html/fixarivan.space" ]] || exit 1

mkdir -p "$BACKUP_PATH"
shopt -s nullglob
backup_files=( "$BACKUP_PATH"/fixarivan_*.tar.gz )
if [[ ${#backup_files[@]} -eq 0 ]]; then
  echo "No backup archive found in $BACKUP_PATH" >&2
  exit 1
fi
LATEST=$(ls -t "${backup_files[@]}" | head -1)
if [[ -z "$LATEST" || ! -f "$LATEST" ]]; then
  echo "Could not resolve latest backup" >&2
  exit 1
fi
[[ "$LATEST" =~ ^/home/fixawcab/backups/fixarivan_.+\.tar\.gz$ ]] || { echo "Unsafe backup path" >&2; exit 1; }

mkdir -p "$TMP_RESTORE"
find "$TMP_RESTORE" -mindepth 1 -maxdepth 1 -exec rm -rf {} + 2>/dev/null || true
if ! tar -xzf "$LATEST" -C "$TMP_RESTORE"; then
  echo "Extraction failed — current site was NOT modified." >&2
  exit 1
fi
if [[ ! -d "$TMP_RESTORE/fixarivan.space" ]]; then
  echo "Extracted tree missing fixarivan.space — aborting, site untouched." >&2
  find "$TMP_RESTORE" -mindepth 1 -maxdepth 1 -exec rm -rf {} + 2>/dev/null || true
  exit 1
fi

# Extraction succeeded — replace live site
mkdir -p "$REMOTE_PATH"
find "$REMOTE_PATH" -mindepth 1 -maxdepth 1 -exec rm -rf {} +
if command -v rsync >/dev/null 2>&1; then
  rsync -a --delete "$TMP_RESTORE/fixarivan.space/" "$REMOTE_PATH/"
else
  cp -a "$TMP_RESTORE/fixarivan.space/." "$REMOTE_PATH/"
fi
  find "$TMP_RESTORE" -mindepth 1 -maxdepth 1 -exec rm -rf {} + 2>/dev/null || true
REMOTE_RB

  ok "Rollback complete"
}

# =============================================================================
# Main
# =============================================================================

main() {
  parse_args "$@"

  if [[ "$ROLLBACK" -eq 1 ]]; then
    validate_remote_dir_constant
    rollback
    return $?
  fi

  validate_remote_dir_constant
  validate

  if [[ "$DRY_RUN" -eq 1 ]]; then
    step "DRY RUN MODE"
  fi

  check_connection
  backup
  clean
  upload
  finalize

  if [[ "$DRY_RUN" -eq 1 ]]; then
    ok "Dry run finished (no remote changes)."
  fi
}

if [[ "${BASH_SOURCE[0]}" == "${0}" ]]; then
  if main "$@"; then
    exit 0
  else
    err "Operation failed."
    exit 1
  fi
fi

END deploy.sh
